import { iniciarJuego, comprobar } from "./js/juegos-master.js"; window.onload = () => { const params = new URLSearchParams(window.location.search); const materia = params.get("materia"); const nivel = params.get("nivel"); const key = materia + nivel; document.getElementById("tituloJuego").innerText = materia.toUpperCase() + " - Nivel " + nivel; iniciarJuego(key); document.getElementById("btnComprobar").onclick = comprobar; };
